GitHub Actions Marketplace
Добавить в CI
# .github/workflows/archlint.yml
name: archlint
on: [push, pull_request]
jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: mshogin/archlint-action@v1
